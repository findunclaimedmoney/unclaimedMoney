import requests
import csv
import json

API_URL = "https://missingcash.com.au/api/import"

def import_csv(filepath):
    with open(filepath, 'r') as f:
        data = f.read()
    
    res = requests.post(API_URL, data=data, headers={"Content-Type": "text/csv"})
    print(json.dumps(res.json(), indent=2))

def import_json(records):
    res = requests.post(API_URL, json=records, headers={"Content-Type": "application/json"})
    print(json.dumps(res.json(), indent=2))
